rules_version = '2'; service cloud.firestore { match /databases/{database}/documents { // Вспомогательная функция: проверяет, авторизован ли пользователь function isSignedIn() { return request.auth != null; } // Вспомогательная функция: получает роль пользователя из коллекции 'users' // Если пользователь не найден, возвращает 'guest' function getUserRole(userId) { return exists(/databases/$(database)/documents/artifacts/$(appId)/public/data/users/$(userId)) ? get(/databases/$(database)/documents/artifacts/$(appId)/public/data/users/$(userId)).data.role : 'guest'; } // Вспомогательная функция: проверяет, является ли пользователь администратором function isAdmin() { return isSignedIn() && getUserRole(request.auth.uid) == 'admin'; } // Вспомогательная функция: проверяет, является ли пользователь начальником function isBoss() { return isSignedIn() && getUserRole(request.auth.uid) == 'boss'; } // Вспомогательная функция: проверяет, является ли пользователь работягой (или выше) function isWorker() { let role = getUserRole(request.auth.uid); return isSignedIn() && (role == 'admin' || role == 'boss' || role == 'worker'); } // Вспомогательная функция: проверяет, является ли пользователь начальником или администратором function isBossOrAdmin() { return isSignedIn() && (getUserRole(request.auth.uid) == 'admin' || getUserRole(request.auth.uid) == 'boss'); } // Правила для коллекции пользователей (users) match /artifacts/{appId}/public/data/users/{userId} { // Чтение разрешено для всех авторизованных пользователей allow read: if isSignedIn(); // Создание разрешено: // 1. Для администраторов // 2. Для самого пользователя, если он регистрируется как 'guest' // и его Telegram ID является чисто цифровым. allow create: if isAdmin() || (request.auth.uid == userId && request.resource.data.role == 'guest' && userId.matches('^\\d+$')); // Проверка, что Telegram ID - это только цифры // Обновление разрешено: // 1. Для администраторов // 2. Для самого пользователя, если он обновляет только свое имя allow update: if isAdmin() || (request.auth.uid == userId && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['name'])); // Удаление разрешено только администраторам allow delete: if isAdmin(); } // Правила для коллекции месячных данных (months) match /artifacts/{appId}/public/data/months/{monthId} { // Чтение разрешено для всех авторизованных пользователей allow read: if isSignedIn(); // Создание и обновление разрешено администраторам, начальникам и работягам allow create, update: if isAdmin() || isBoss() || isWorker(); // Удаление разрешено только администраторам allow delete: if isAdmin(); } // Правила для коллекции уведомлений (notifications) // Разрешено создание только администраторам // Чтение, обновление и удаление запрещено match /artifacts/{appId}/public/data/notifications/{document=**} { allow create: if isAdmin(); allow read, update, delete: if false; } } }
